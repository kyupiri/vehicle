Apps Script 설정

스프레드시트에서 확장 프로그램 → Apps Script를 열고, 아래 코드를 붙여넣고 저장(Ctrl+S)하세요.

const SHEET_NAME = '사용일지'; function doPost(e) { const sheet = SpreadsheetApp.getActiveSpreadsheet().getSheetByName(SHEET_NAME); const data = JSON.parse(e.postData.contents); if (sheet.getLastRow() === 0) { sheet.appendRow(['날짜','사용자','동승자','목적지','사용목적','사용전km','사용후km','주행거리','비고','등록시각']); } sheet.appendRow([ data.date, data.driver, data.passengers, data.destination, data.purpose, data.kmBefore, data.kmAfter, data.kmDiff, data.note, new Date().toLocaleString('ko-KR') ]); return ContentService .createTextOutput(JSON.stringify({result:'success'})) .setMimeType(ContentService.MimeType.JSON); } function doGet(e) { const sheet = SpreadsheetApp.getActiveSpreadsheet().getSheetByName(SHEET_NAME); const rows = sheet.getDataRange().getValues(); const headers = rows[0]; const records = rows.slice(1).map(r => { const obj = {}; headers.forEach((h,i) => obj[h] = r[i]); return obj; }); return ContentService .createTextOutput(JSON.stringify(records)) .setMimeType(ContentService.MimeType.JSON); } function deleteRow(e) { const sheet = SpreadsheetApp.getActiveSpreadsheet().getSheetByName(SHEET_NAME); const idx = parseInt(e.parameter.row); sheet.deleteRow(idx + 2); return ContentService .createTextOutput(JSON.stringify({result:'deleted'})) .setMimeType(ContentService.MimeType.JSON); }

웹 앱으로 배포

Apps Script에서 배포 → 새 배포를 클릭하고, 유형을 웹 앱으로 선택하세요.
"다음 사용자로 실행"은 나, "액세스 권한"은 모든 사용자로 설정 후 배포하세요.
